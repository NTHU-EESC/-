#include<stdio.h>
int main(void)
{
char map[10][10];
int  R,C,N,r,i,j,m;
scanf("%d ",&N);
scanf("%d %d",&R,&C);
scanf("%d",&r);


for(i=0;i<N;i++){
    for(j=0;j<N;j++){
       map[i][j]='-';

    }

}
i=R-1;
j=C-1;
for(m=1;m<=r;m++){
        if(i+m>N-1)
            break;
       else
       map[i+m][j]='+';
}
for(m=1;m<=r;m++){
        if(i-m<0)
            break;
        else
        map[i-m][j]='+';
}
for(m=1;m<=r;m++){
        if(j+m>N-1)
            break;
        else
        map[i][j+m]='+';
}
for(m=1;m<=r;m++){
        if(j-m<0)
         break;
        else
        map[i][j-m]='+';

}

map[i][j]='S';
	for(i=0;i<N;i++){
		for(j=0;j<N;j++){
			printf("%c",map[i][j]);
		}
		printf("\n");
	}
return 0;



}
