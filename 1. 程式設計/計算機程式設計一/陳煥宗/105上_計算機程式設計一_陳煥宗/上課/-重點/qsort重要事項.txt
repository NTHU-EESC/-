#include <stdio.h>
#include <stdlib.h> //ずΤqsort, rand(), RAND_MAX
#include <string.h>
#define SIZE 10

int compare_int (const void *a, const void *b)  //constTO把计肚iㄓぃ|Q拽->ぃ喑Qэ->ぃ喟++or--单单
{
    const int *va = (const int *) a;
    const int *vb = (const int *) b;
    return *va-*vb;
    //return *((const int *)a) - *((const int *)b);  //ョiHрW堡T媛菠痞é
}

int compare_double (const void *a, const void *b)  //夹非gk
{
    const double *da = (const double *) a;
    const double *db = (const double *) b;
    return (*da > *db) - (*da < *db);  //(*da > *db)P(*da < *db)OP_ΑAだOu|^肚1┪O0
    /*if(*da>*db) return 1;
    else if(*da<*db) return -1;
    else return 0;*/                   //ョio妓g
}

int main(void)
{
    int data1[SIZE];
    double data2[SIZE];
    int i;

    for (i=0; i<SIZE; i++) {
        data1[i] = rand()%SIZE;
        data2[i] = (double) rand()/RAND_MAX;  //***рrand()锣传Θdouble
    }

    printf("original: ");
    for (i=0; i<SIZE; i++) {
        printf("%d ", data1[i]);
    }
    printf("\n");

    printf("  sorted: ");
    qsort(data1, SIZE, sizeof(int), compare_int);  //***
    for (i=0; i<SIZE; i++) {
        printf("%d ", data1[i]);
    }
    printf("\n");

    printf("original: ");
    for (i=0; i<SIZE; i++) {
        printf("%.2f ", data2[i]);
    }
    printf("\n");

    printf("  sorted: ");
    qsort(data2, SIZE, sizeof(double), compare_double);
    for (i=0; i<SIZE; i++) {
        printf("%.2f ", data2[i]);
    }
    printf("\n");

    return 0;
}
